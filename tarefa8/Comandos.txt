show tables;
create database 'pizzaria';
use pizzaria;


CREATE TABLE CLIENTE{
CODCLIENTE INTEGER NOT NULL,
NOME VARCHAR (60) NOT NULL,
SOBRENOME VARCHAR (60) NOT NULL,
PREFERENCIA VARCHAR (60) NOT NULL,
TELEFONE INTEGER (20) NOT NULL,
SEXO CHAR(01) CHECK (SEXO = `F´ OR SEXO = `M´),
BAIRRO VARCHAR (30),
RUA VARCHAR (30),
CIDADE VARCHAR(50) NOT NULL,
CLIENTE_ANTIGO CHAR(01) CHECK (CLIENTE_ANTIGO = `N´ OR CLIENTE_ANTIGO = `S´),
PRIMARY KEY (CODCLIENTE)
};

CREATE TABLE PIZZA{
CODPIZZA INTEGER NOT NULL,
NOME VARCHAR (60) NOT NULL,
SABOR VARCHAR (30) NOT NULL,
PRECO FLOAT (20) NOT NULL,
PRIMARY KEY (CODPIZZA),
FOREIGN KEY (CODPIZZA) REFERENCES CLIENTE (CODCLIENTE)
};

CREATE TABLE LOJAS{
NOME VARCHAR (60) NOT NULL,
TELEFONE INTEGER (20) NOT NULL,
BAIRRO VARCHAR (30) NOT NULL,
RUA VARCHAR (30) NOT NULL,
CIDADE VARCHAR(50) NOT NULL,
ESTADO VARCHAR(50) NOT NULL,
PRIMARY KEY (NOME)
};

CREATE FUNCIONARIOS{
NOME VARCHAR (30) NOT NULL,
TELEFONE INTEGER (20) NOT NULL,
SOBRENOME VARCHAR (30) NOT NULL,
FUNCAO VARCHAR (30) NOT NULL,
SALARIO FLOAT (20) NOT NULL,
PRIMARY KEY (NOME)
};

CREATE DESPESAS{
AGUA FLOAT (30) NOT NULL,
LUZ FLOAT (30) NOT NULL,
TELEFONE VARCHAR (30) NOT NULL,
IPTU FLOAT (30) NOT NULL,
IMPOSTOS1 FLOAT (100) NOT NULL,
IMPOSTOS2 FLOAT (100) NOT NULL,
OUTROS1 FLOAT (100) NOT NULL,
OUTROS2 FLOAT (100) NOT NULL
};

CREATE FORNECEDORES{
NOME VARCHAR (30) NOT NULL,
TELEFONE INTEGER (20) NOT NULL,
BAIRRO VARCHAR (30) NOT NULL,
RUA VARCHAR (30) NOT NULL,
CIDADE VARCHAR (30) NOT NULL,
ESTADO VARCHAR (30) NOT NULL,
PRIMARY KEY (NOME)
};
